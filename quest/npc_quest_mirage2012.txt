function	script	sekiban_func	{
	for(set '@i,1; '@i<=10; set '@i,'@i+1){
		set '@progress$,(10*getarg(0)/getarg(1)<'@i ? "■":"□")+'@progress$;
	}
	set '@num,200 - 200*getarg(0)/getarg(1);
	mes "[不思議な石版]";
	if('@num)
		mes "ウィィィイイイイン……";
	else
		mes "ファァァァーーーン……";
	mes "　";
	mes "‐不思議な文字が描かれている‐";
	mes '@progress$;
	mes "…… " +'@num;
	close;
	return;
}
